def finalize
  if @server

    @server.close
    @server = nil
  end
end

def unicast(socket, payload)
  socket.write(payload)
  true
rescue IOError, Errno::ECONNRESET, Errno::EPIPE
  @sockets.delete(socket)
  false
end
