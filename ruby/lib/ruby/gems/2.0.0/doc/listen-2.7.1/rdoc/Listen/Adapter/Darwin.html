def _changes_path(changes)
  changes.map do |path|
    path.sub!(/\/$/, '')
    Pathname.new(path)
  end
end

def _init_worker
  FSEvent.new.tap do |worker|
    worker.watch(_directories_path, latency: _latency) do |changes|
      _changes_path(changes).each { |path| _notify_change(path, type: 'Dir') }
    end
  end
end
