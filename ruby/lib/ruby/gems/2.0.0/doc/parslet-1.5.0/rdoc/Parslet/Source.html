def initialize(str)
  raise ArgumentError unless str.respond_to?(:to_str)

  @pos = 0
  @str = str
  
  @line_cache = LineCache.new
  @line_cache.scan_for_line_endings(0, @str)
end

def consume(n)
  slice_str = @str.slice(@pos, n)
  slice = Parslet::Slice.new(
    slice_str, 
    pos,
    @line_cache)
  
  @pos += slice_str.size
  return slice
end
