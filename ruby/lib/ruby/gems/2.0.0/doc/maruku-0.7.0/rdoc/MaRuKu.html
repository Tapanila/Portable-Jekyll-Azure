    FootnoteText

    %r{

    ^  # begin of line
[ ]{0,3} # up to 3 spaces
\* # one asterisk
\[ # opening bracket
([^\]]+) # any non-closing bracket:  id = $1
\] # closing bracket
:  # colon
\s* # whitespace
(\S.*\S)* #           definition=$2
\s* # strip this whitespace
$   # end of line

    }x

    IncompleteLink

    %r{

    ^[ ]{0,3}\[([^\[\]]+)\]:    # id = $1
  [ ]*
<?([^>\s]+)>?       # url = $2
  [ ]*
(?: # Titles are delimited by "quotes" or (parens).
  (?:(?:"([^"]+)")|(?:'([^']+)')|(?:\(([^\(\)]+)\))) # title = $3, $4, or $5
  \s*(.+)? # stuff = $6
)?  # title is optional

    }x
