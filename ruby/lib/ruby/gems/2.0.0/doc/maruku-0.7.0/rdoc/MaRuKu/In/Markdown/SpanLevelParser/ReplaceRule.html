    Sort of like “join” - places the elements in “elem” between each adjacent element in the array.

    # File lib/maruku/input/rubypants.rb, line 176
def intersperse(ary, elem)
  return ary if ary.length <= 1
  h, *t = ary
  t.inject([h]) do |r, e|
    entities = elem.map do |el|
      en = el.clone
      en.doc = doc
      en
    end
    r.concat entities
    r << e
  end
end
