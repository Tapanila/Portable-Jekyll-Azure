def fire(now = Time.now)
  time = now + 0.001 # Fudge 1ms in case of clock imprecision
  while (timer = @timers.first) && (time >= timer.time)
    @timers.delete timer
    timer.fire(now)
  end
end

def wait_interval(now = Time.now)
  timer = @timers.first
  return unless timer
  interval = timer.time - now
  interval > 0 ? interval : 0
end
